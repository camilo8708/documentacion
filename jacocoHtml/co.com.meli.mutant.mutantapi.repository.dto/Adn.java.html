package co.com.meli.mutant.mutantapi.repository.dto;

import org.springframework.data.annotation.Id;

/**
 * Clase para representar la entidad que almacena los resultados de las validaciones de las cadenas de ADN
 */
public class Adn {
    @Id
    private String dna;
    private boolean isMutant;

    public Adn() {
    }

    public Adn(String dna, boolean isMutant) {
        this.dna = dna;
        this.isMutant = isMutant;
    }

    public String getDna() {
        return dna;
    }

    public void setDna(String dna) {
        this.dna = dna;
    }

    public boolean isMutant() {
        return isMutant;
    }

    public void setMutant(boolean mutant) {
        isMutant = mutant;
    }

    @Override
    public String toString() {
        return "Adn{" +
                "dna='" + dna + '\'' +
                '}';
    }
}
